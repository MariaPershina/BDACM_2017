    \node[rectangle, draw=black, below of = eta] (mu)     {expected value \\

    \node[rectangle, draw=gray, right of = mu, node distance = 5cm] (thetaLH)     {\textcolor{gray}{parameters
      likelihood} \\
    $\textcolor{gray}{\theta_{LH} \sim \text{some prior}}$};

    \node[rectangle, draw=black, below of = mu] (LH)     {likelihood of data \\
    $y \sim LH(\mu, \textcolor{gray}{\theta_{LH}})$};

    \draw[->] (beta)--(eta);

    \draw[->] (x)--(eta);

    \draw[->] (eta)--(mu) node [ pos=0.5, left] {(inverse) link function};

    \draw[->, color = gray] (thetaLF)--(mu);

    \draw[->] (mu)--(LH) node [ pos=0.5, left] {likelihood function};

    \draw[->, color = gray] (thetaLH)--(LH);
